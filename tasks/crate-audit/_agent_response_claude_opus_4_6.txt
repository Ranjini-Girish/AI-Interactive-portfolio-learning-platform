#[derive(Clone, Debug, Eq, PartialEq)]
struct Version {
    major: u64,
    minor: u64,
    patch: u64,
    pre: Vec<PreRelaseId>,
}

#[derive(Clone, Debug, Eq, PartialEq)]
enum PreRelaseId {
    Num(u64),
    Str(String),
}

impl Ord for PreRelaseId {
    fn cmp(&self, other: &Self) -> std::cmp::Ordering {
        match (self, other) {
            (PreRelaseId::Num(a), PreRelaseId::Num(b)) => a.cmp(b),
            (PreRelaseId::Str(a), PreRelaseId::Str(b)) => a.cmp(b),
            (PreRelaseId::Num(_), PreRelaseId::Str(_)) => std::cmp::Ordering::Less,
            (PreRelaseId::Str(_), PreRelaseId::Num(_)) => std::cmp::Ordering::Greater,
        }
    }
}

impl PartialOrd for PreRelaseId {
    fn partial_cmp(&self, other: &Self) -> Option<std::cmp::Ordering> {
        Some(self.cmp(other))
    }
}

impl Ord for Version {
    fn cmp(&self, other: &Self) -> std::cmp::Ordering {
        match self.major.cmp(&other.major) {
            std::cmp::Ordering::Equal => {}
            o => return o,
        }
        match self.minor.cmp(&other.minor) {
            std::cmp::Ordering::Equal => {}
            o => return o,
        }
        match self.patch.cmp(&other.patch) {
            std::cmp::Ordering::Equal => {}
            o => return o,
        }
        // No pre-release > any pre-release
        match (self.pre.is_empty(), other.pre.is_empty()) {
            (true, true) => std::cmp::Ordering::Equal,
            (true, false) => std::cmp::Ordering::Greater,
            (false, true) => std::cmp::Ordering::Less,
            (false, false) => {
                for (a, b) in self.pre.iter().zip(other.pre.iter()) {
                    match a.cmp(b) {
                        std::cmp::Ordering::Equal => continue,
                        o => return o,
                    }
                }
                self.pre.len().cmp(&other.pre.len())
            }
        }
    }
}

impl PartialOrd for Version {
    fn partial_cmp(&self, other: &Self) -> Option<std::cmp::Ordering> {
        Some(self.cmp(other))
    }
}

impl Version {
    fn parse(s: &str) -> Version {
        let s = s.trim();
        let (main, pre_str) = if let Some(idx) = s.find('-') {
            (&s[..idx], Some(&s[idx + 1..]))
        } else {
            (s, None)
        };
        let parts: Vec<&str> = main.split('.').collect();
        let major = parts[0].parse().unwrap();
        let minor = parts[1].parse().unwrap();
        let patch = parts[2].parse().unwrap();
        let pre = if let Some(ps) = pre_str {
            ps.split('.')
                .map(|id| {
                    if let Ok(n) = id.parse::<u64>() {
                        PreRelaseId::Num(n)
                    } else {
                        PreRelaseId::Str(id.to_string())
                    }
                })
                .collect()
        } else {
            vec![]
        };
        Version { major, minor, patch, pre }
    }

    fn to_string(&self) -> String {
        let base = format!("{}.{}.{}", self.major, self.minor, self.patch);
        if self.pre.is_empty() {
            base
        } else {
            let pre_str: Vec<String> = self.pre.iter().map(|p| match p {
                PreRelaseId::Num(n) => n.to_string(),
                PreRelaseId::Str(s) => s.clone(),
            }).collect();
            format!("{}-{}", base, pre_str.join("."))
        }
    }
}

// ---- Constraints ----

#[derive(Clone, Debug)]
enum Constraint {
    Exact(Version),
    Gte(Version),
    Gt(Version),
    Lte(Version),
    Lt(Version),
    Caret(Version),
    Tilde(Version),
}

fn parse_constraints(s: &str) -> Vec<Constraint> {
    let s = s.trim();
    let parts: Vec<&str> = s.split(',').collect();
    parts.iter().map(|p| parse_single_constraint(p.trim())).collect()
}

fn parse_single_constraint(s: &str) -> Constraint {
    let s = s.trim();
    if s.starts_with(">=") {
        Constraint::Gte(Version::parse(&s[2..]))
    } else if s.starts_with('>') {
        Constraint::Gt(Version::parse(&s[1..]))
    } else if s.starts_with("<=") {
        Constraint::Lte(Version::parse(&s[2..]))
    } else if s.starts_with('<') {
        Constraint::Lt(Version::parse(&s[1..]))
    } else if s.starts_with('=') {
        Constraint::Exact(Version::parse(&s[1..]))
    } else if s.starts_with('^') {
        Constraint::Caret(Version::parse(&s[1..]))
    } else if s.starts_with('~') {
        Constraint::Tilde(Version::parse(&s[1..]))
    } else {
        Constraint::Exact(Version::parse(s))
    }
}

fn version_satisfies(v: &Version, constraint: &Constraint) -> bool {
    match constraint {
        Constraint::Exact(target) => v == target,
        Constraint::Gte(target) => v >= target,
        Constraint::Gt(target) => v > target,
        Constraint::Lte(target) => v <= target,
        Constraint::Lt(target) => v < target,
        Constraint::Caret(target) => {
            // >=target, < next breaking
            if v < target {
                return false;
            }
            if target.major != 0 {
                // <(major+1).0.0
                v.major == target.major
                    && Version { major: v.major, minor: v.minor, patch: v.patch, pre: v.pre.clone() }
                        < Version { major: target.major + 1, minor: 0, patch: 0, pre: vec![] }
            } else if target.minor != 0 {
                // ^0.Y.Z => >=0.Y.Z, <0.(Y+1).0
                v.major == 0 && v.minor == target.minor
                    && *v < Version { major: 0, minor: target.minor + 1, patch: 0, pre: vec![] }
            } else {
                // ^0.0.Z => >=0.0.Z, <0.0.(Z+1)
                v.major == 0 && v.minor == 0 && v.patch == target.patch
                    && *v < Version { major: 0, minor: 0, patch: target.patch + 1, pre: vec![] }
            }
        }
        Constraint::Tilde(target) => {
            // >=target, <major.(minor+1).0
            if v < target {
                return false;
            }
            v.major == target.major && v.minor == target.minor
                && *v < Version { major: target.major, minor: target.minor + 1, patch: 0, pre: vec![] }
        }
    }
}

fn satisfies_all(v: &Version, constraints: &[Constraint]) -> bool {
    constraints.iter().all(|c| version_satisfies(v, c))
}

#[derive(Deserialize, Debug)]
struct RegistryPackage {
    name: String,
    versions: Vec<RegistryVersion>,
}

#[derive(Deserialize, Debug)]
struct RegistryVersion {
    version: String,
    dependencies: HashMap<String, String>,
    license: String,
    yanked: bool,
    vulnerability_score: f64,
}

#[derive(Deserialize, Debug)]
struct Project {
    name: String,
    dependencies: HashMap<String, String>,
}

#[derive(Deserialize, Debug)]
struct Config {

fn main() {
    // Load config
    let config: Config = serde_json::from_str(
        &fs::read_to_string("/app/data/config.json").unwrap()
    ).unwrap();

    // Load project
    let project: Project = serde_json::from_str(
        &fs::read_to_string("/app/data/project.json").unwrap()
    ).unwrap();

    // Load registry
    let mut registry: HashMap<String, RegistryPackage> = HashMap::new();
    let registry_dir = Path::new("/app/data/registry/");
    for entry in fs::read_dir(registry_dir).unwrap() {
        let entry = entry.unwrap();
        let path = entry.path();
        if path.extension().map_or(false, |e| e == "json") {
            let content = fs::read_to_string(&path).unwrap();
            let pkg: RegistryPackage = serde_json::from_str(&content).unwrap();
            registry.insert(pkg.name.clone(), pkg);
        }
    }

    // Build license category map
    let mut license_to_category: HashMap<String, String> = HashMap::new();
    for (cat, licenses) in &config.license_categories {
        for lic in licenses {
            license_to_category.insert(lic.clone(), cat.clone());
        }
    }

    // Resolve dependencies
    // resolved: name -> (Version, resolved version info)
    struct ResolvedPkg {
        name: String,
        version: Version,
        version_str: String,
        license: String,
        vulnerability_score: f64,
        dependencies: HashMap<String, String>, // name -> constraint string
    }

    let mut resolved: HashMap<String, ResolvedPkg> = HashMap::new();
    // constraints accumulated per package: name -> Vec<Constraint>
    let mut all_constraints: HashMap<String, Vec<Constraint>> = HashMap::new();
    // Queue of (package_name, constraint_string)
    let mut queue: VecDeque<(String, String)> = VecDeque::new();

    // Add project's direct dependencies
    for (dep_name, constraint_str) in &project.dependencies {
        queue.push_back((dep_name.clone(), constraint_str.clone()));
    }

    while let Some((pkg_name, constraint_str)) = queue.pop_front() {
        let new_constraints = parse_constraints(&constraint_str);

        // Add new constraints
        let entry = all_constraints.entry(pkg_name.clone()).or_insert_with(Vec::new);
        let had_constraints_before = !entry.is_empty();
        entry.extend(new_constraints);

        let all_cons = &all_constraints[&pkg_name];

        // Find highest non-yanked version satisfying all constraints
        let reg_pkg = registry.get(&pkg_name).unwrap_or_else(|| panic!("Package {} not found in registry", pkg_name));
        
        let mut candidates: Vec<&RegistryVersion> = reg_pkg.versions.iter()
            .filter(|v| !v.yanked)
            .filter(|v| {
                let ver = Version::parse(&v.version);
                satisfies_all(&ver, all_cons)
            })
            .collect();
        
        candidates.sort_by(|a, b| {
            let va = Version::parse(&a.version);
            let vb = Version::parse(&b.version);
            vb.cmp(&va)
        });

        let best = candidates.first().unwrap_or_else(|| panic!("No version found for {} with constraints", pkg_name));
        let best_version = Version::parse(&best.version);

        if let Some(existing) = resolved.get(&pkg_name) {
            if existing.version == best_version {
                // Same version, no change needed
                continue;
            }
            // Version changed due to new constraints - need to re-resolve
        }

        let rpkg = ResolvedPkg {
            name: pkg_name.clone(),
            version: best_version.clone(),
            version_str: best.version.clone(),
            license: best.license.clone(),
            vulnerability_score: best.vulnerability_score,
            dependencies: best.dependencies.clone(),
        };

        resolved.insert(pkg_name.clone(), rpkg);

        // Add transitive dependencies to queue
        for (dep_name, dep_constraint) in &best.dependencies {
            queue.push_back((dep_name.clone(), dep_constraint.clone()));
        }
    }

    // Build dependency graph
    // For each resolved package, its direct dependencies (also resolved)
    let pkg_names: Vec<String> = resolved.keys().cloned().collect();
    let mut dep_graph: HashMap<String, Vec<String>> = HashMap::new(); // pkg -> [deps]
    let mut reverse_graph: HashMap<String, Vec<String>> = HashMap::new(); // pkg -> [dependents]

    for name in &pkg_names {
        dep_graph.insert(name.clone(), Vec::new());
        reverse_graph.insert(name.clone(), Vec::new());
    }

    for (name, rpkg) in &resolved {
        for dep_name in rpkg.dependencies.keys() {
            if resolved.contains_key(dep_name) {
                dep_graph.get_mut(name).unwrap().push(dep_name.clone());
                reverse_graph.get_mut(dep_name).unwrap().push(name.clone());
            }
        }
    }

    // Sort dependency lists for determinism
    for v in dep_graph.values_mut() {
        v.sort();
    }
    for v in reverse_graph.values_mut() {
        v.sort();
    }

    // Compute depths: longest path from project root
    // Project root connects to its direct dependencies
    // Depth = longest path from root to package
    // We need to compute longest path in a DAG
    let mut depths: HashMap<String, usize> = HashMap::new();
    
    // BFS/DFS from project root
    // Project's direct deps are at depth >= 1
    // But depth is longest path, so we need to compute it properly
    
    // Add project as a virtual node with edges to its direct deps
    // Then compute longest path from project to each node
    // Project itself has depth 0, its direct deps have depth >= 1
    
    // Topological sort first (of the resolved packages + project)
    // Then compute longest paths
    
    // Build full graph including project
    let project_name = &project.name;
    let mut full_dep_graph: HashMap<String, Vec<String>> = dep_graph.clone();
    let mut project_deps: Vec<String> = Vec::new();
    for dep_name in project.dependencies.keys() {
        if resolved.contains_key(dep_name) {
            project_deps.push(dep_name.clone());
        }
    }
    project_deps.sort();
    full_dep_graph.insert(project_name.clone(), project_deps.clone());

    // Topological sort using Kahn's algorithm on the full graph
    let mut in_degree: HashMap<String, usize> = HashMap::new();
    for name in full_dep_graph.keys() {
        in_degree.entry(name.clone()).or_insert(0);
    }
    for (_, deps) in &full_dep_graph {
        for dep in deps {
            *in_degree.entry(dep.clone()).or_insert(0) += 1;
        }
    }

    // Compute longest path from project root
    // Initialize all depths to 0
    let mut dist: HashMap<String, usize> = HashMap::new();
    for name in full_dep_graph.keys() {
        dist.insert(name.clone(), 0);
    }
    dist.insert(project_name.clone(), 0);

    // Process in topological order (from sources to sinks)
    // But we want longest path from project, so we process in topo order
    // and relax edges
    
    // Topo sort
    let mut topo_order: Vec<String> = Vec::new();
    let mut in_deg = in_degree.clone();
    let mut q: BTreeSet<String> = BTreeSet::new();
    for (name, &deg) in &in_deg {
        if deg == 0 {
            q.insert(name.clone());
        }
    }
    while let Some(name) = q.iter().next().cloned() {
        q.remove(&name);
        topo_order.push(name.clone());
        if let Some(deps) = full_dep_graph.get(&name) {
            for dep in deps {
                let d = in_deg.get_mut(dep).unwrap();
                *d -= 1;
                if *d == 0 {
                    q.insert(dep.clone());
                }
            }
        }
    }

    // Compute longest path from project_name
    for name in &topo_order {
        if let Some(deps) = full_dep_graph.get(name) {
            let current_dist = dist[name];
            for dep in deps {
                let new_dist = current_dist + 1;
                if new_dist > dist[dep] {
                    dist.insert(dep.clone(), new_dist);
                }
            }
        }
    }

    for (name, d) in &dist {
        if name != project_name {
            depths.insert(name.clone(), *d);
        }
    }

    // Build order: topological sort of resolved packages only (not project)
    // Packages with no deps first, alphabetical tie-breaking
    let mut build_order: Vec<String> = Vec::new();
    let mut build_in_deg: HashMap<String, usize> = HashMap::new();
    for name in &pkg_names {
        build_in_deg.insert(name.clone(), 0);
    }
    for (name, deps) in &dep_graph {
        for dep in deps {
            *build_in_deg.get_mut(dep).unwrap() += 1;
        }
    }
    let mut build_q: BTreeSet<String> = BTreeSet::new();
    for (name, &deg) in &build_in_deg {
        if deg == 0 {
            build_q.insert(name.clone());
        }
    }
    while let Some(name) = build_q.iter().next().cloned() {
        build_q.remove(&name);
        build_order.push(name.clone());
        if let Some(deps) = dep_graph.get(&name) {
            for dep in deps {
                // Wait, build order: a package must be built before its dependents
                // So we need reverse: if A depends on B, B must be built first
                // In dep_graph, A -> [B, C] means A depends on B and C
                // For build order, we want B and C before A
                // So we should use reverse_graph for Kahn's
            }
        }
    }

    // Redo build order correctly
    // For build order: if A depends on B, then B must come before A
    // In Kahn's: in-degree based on "who depends on me" = reverse_graph edges count
    // Actually, let me think again:
    // dep_graph: A -> [B, C] means A depends on B and C
    // For build order, B and C must come before A
    // So the edge direction for topo sort should be: B -> A, C -> A (i.e., "B must be before A")
    // In-degree of A = number of dependencies A has
    
    build_order.clear();
    let mut build_in_deg2: HashMap<String, usize> = HashMap::new();
    for name in &pkg_names {
        build_in_deg2.insert(name.clone(), dep_graph[name].len());
    }
    let mut build_q2: BTreeSet<String> = BTreeSet::new();
    for (name, &deg) in &build_in_deg2 {
        if deg == 0 {
            build_q2.insert(name.clone());
        }
    }
    while let Some(name) = build_q2.iter().next().cloned() {
        build_q2.remove(&name);
        build_order.push(name.clone());
        // For each package that depends on 'name', decrease their in-degree
        if let Some(dependents) = reverse_graph.get(&name) {
            for dependent in dependents {
                let d = build_in_deg2.get_mut(dependent).unwrap();
                *d -= 1;
                if *d == 0 {
                    build_q2.insert(dependent.clone());
                }
            }
        }
    }

    // Compute effective vulnerability scores
    // effective_score(pkg) = base_score + decay * sum(effective_score(dep) for dep in deps)
    // Process in build order (dependencies first)
    let mut effective_scores: HashMap<String, f64> = HashMap::new();
    for name in &build_order {
        let rpkg = &resolved[name];
        let base = rpkg.vulnerability_score;
        let mut dep_sum = 0.0;
        for dep in &dep_graph[name] {
            dep_sum += effective_scores[dep];
        }
        let eff = base + config.vulnerability_decay_factor * dep_sum;
        effective_scores.insert(name.clone(), eff);
    }

    // Build resolved packages output
    let mut resolved_list: Vec<Value> = Vec::new();
    let mut sorted_names: Vec<String> = pkg_names.clone();
    sorted_names.sort();

    // Sort by depth ascending, then name ascending
    let mut name_depth: Vec<(String, usize)> = pkg_names.iter()
        .map(|n| (n.clone(), depths[n]))
        .collect();
    name_depth.sort_by(|a, b| a.1.cmp(&b.1).then(a.0.cmp(&b.0)));

    for (name, depth) in &name_depth {
        let rpkg = &resolved[name];
        let cat = license_to_category.get(&rpkg.license)
            .cloned()
            .unwrap_or_else(|| "unknown".to_string());
        
        let mut direct_deps: Vec<String> = dep_graph[name].iter()
            .map(|d| format!("{}@{}", d, resolved[d].version_str))
            .collect();
        direct_deps.sort();

        let mut dependents: Vec<String> = Vec::new();
        // Dependents include project root and other resolved packages
        // Check if project depends on this package directly
        if project.dependencies.contains_key(name) {
            dependents.push(format!("{}@0.0.0", project_name));
            // Wait, project doesn't have a version... Let me re-read the spec
            // "dependents" are packages that depend on this one
            // The spec says "excluding the project root" for max_fan_in
            // But for dependents list... let me check output_schema
            // It says dependents: [<string "name@version">, ...]
            // The project doesn't have a version in the manifest
            // Let me check if project root should be included
        }
        
        // Actually, looking at the spec more carefully:
        // "max_fan_in: Maximum number of dependents any single package has (reverse dependencies in the resolved graph, excluding the project root)"
        // So for max_fan_in we exclude project root
        // But for the dependents list in resolved_package, it's not clear
        // Let me include only resolved packages as dependents (not project root)
        
        // Actually wait - the resolved packages are "excluding the project itself" per statistics
        // The dependents should be other resolved packages that depend on this one
        
        let mut dependents: Vec<String> = reverse_graph[name].iter()
            .map(|d| format!("{}@{}", d, resolved[d].version_str))
            .collect();
        dependents.sort();

        resolved_list.push(json!({
            "name": name,
            "version": rpkg.version_str,
            "license": rpkg.license,
            "license_category": cat,
            "depth": depth,
            "direct_dependencies": direct_deps,
            "dependents": dependents
        }));
    }

    // Build order as name@version strings
    let build_order_strs: Vec<String> = build_order.iter()
        .map(|n| format!("{}@{}", n, resolved[n].version_str))
        .collect();

    // Vulnerability report
    let precision = config.output_precision;
    
    let mut vulnerable_entries: Vec<(String, f64, f64)> = Vec::new(); // (name, base, effective)
    let mut max_score: f64 = 0.0;
    let mut total_vulnerable = 0;

    for name in &pkg_names {
        let eff = effective_scores[name];
        if eff > 0.0 {
            total_vulnerable += 1;
            let base = resolved[name].vulnerability_score;
            vulnerable_entries.push((name.clone(), base, eff));
            if eff > max_score {
                max_score = eff;
            }
        }
    }

    // critical_packages sorted by effective_score desc, then name asc
    vulnerable_entries.sort_by(|a, b| {
        b.2.partial_cmp(&a.2).unwrap().then(a.0.cmp(&b.0))
    });

    let critical_packages: Vec<Value> = vulnerable_entries.iter().map(|(name, base, eff)| {
        json!({
            "name": name,
            "version": resolved[name].version_str,
            "base_score": format_float(*base, precision),
            "effective_score": format_float(*eff, precision)
        })
    }).collect();

    // License conflicts
    let category_rank = |cat: &str| -> i32 {
        match cat {
            "permissive" => 0,
            "weak_copyleft" => 1,
            "strong_copyleft" => 2,
            "unknown" => -1, // unknown doesn't cause conflicts? Let me think...
            _ => -1,
        }
    };

    // A conflict occurs when a package depends on another with MORE restrictive license
    // permissive < weak_copyleft < strong_copyleft
    // So if pkg has permissive and dep has strong_copyleft, that's a conflict
    let mut conflicts: Vec<Value> = Vec::new();
    
    for name in &pkg_names {
        let rpkg = &resolved[name];
        let pkg_cat = license_to_category.get(&rpkg.license)
            .cloned()
            .unwrap_or("unknown".to_string());
        let pkg_rank = category_rank(&pkg_cat);
        
        for dep_name in &dep_graph[name] {
            let dep_rpkg = &resolved[dep_name];
            let dep_cat = license_to_category.get(&dep_rpkg.license)
                .cloned()
                .unwrap_or("unknown".to_string());
            let dep_rank = category_rank(&dep_cat);
            
            // Conflict if dependency has MORE restrictive license
            if dep_rank > pkg_rank && pkg_rank >= 0 && dep_rank >= 0 {
                conflicts.push(json!({
                    "package": format!("{}@{}", name, rpkg.version_str),
                    "package_license": rpkg.license,
                    "dependency": format!("{}@{}", dep_name, dep_rpkg.version_str),
                    "dependency_license": dep_rpkg.license
                }));
            }
        }
    }

    // Sort conflicts by package name ascending
    conflicts.sort_by(|a, b| {
        let pa = a["package"].as_str().unwrap();
        let pb = b["package"].as_str().unwrap();
        pa.cmp(pb)
    });

    let total_conflicts = conflicts.len();

    // Statistics
    let total_packages = pkg_names.len();
    let max_depth = depths.values().cloned().max().unwrap_or(0);
    let avg_depth = if total_packages > 0 {
        depths.values().sum::<usize>() as f64 / total_packages as f64
    } else {
        0.0

        } else if p.starts_with("<=") {

fn satisfies(v: &Version, c: &Constraint) -> bool {
    match c {
        Constraint::Exact(t) => v == t,
        Constraint::Gte(t) => v >= t,
        Constraint::Gt(t) => v > t,
        Constraint::Lte(t) => v <= t,
        Constraint::Lt(t) => v < t,
        Constraint::Caret(t) => {
            if v < t { return false; }
            if t.major != 0 {
                v.major == t.major
            } else if t.minor != 0 {
                v.major == 0 && v.minor == t.minor
            } else {

use serde::Serialize;
use serde_json::{json, Value};
use std::collections::{BTreeMap, BTreeSet, HashMap, VecDeque};
use std::fs;

#[derive(Clone, Debug, Eq, PartialEq)]
struct Ver {
    major: u64,
    minor: u64,
    patch: u64,
    pre: Vec<PreId>,
}

#[derive(Clone, Debug, Eq, PartialEq)]
enum PreId {
    Num(u64),
    Str(String),
}

impl Ord for PreId {
    fn cmp(&self, other: &Self) -> std::cmp::Ordering {
        match (self, other) {
            (PreId::Num(a), PreId::Num(b)) => a.cmp(b),
            (PreId::Str(a), PreId::Str(b)) => a.cmp(b),
            (PreId::Num(_), PreId::Str(_)) => std::cmp::Ordering::Less,
            (PreId::Str(_), PreId::Num(_)) => std::cmp::Ordering::Greater,
        }
    }
}
impl PartialOrd for PreId {
    fn partial_cmp(&self, other: &Self) -> Option<std::cmp::Ordering> { Some(self.cmp(other)) }
}

impl Ord for Ver {
    fn cmp(&self, other: &Self) -> std::cmp::Ordering {
        let c = self.major.cmp(&other.major)
            .then(self.minor.cmp(&other.minor))
            .then(self.patch.cmp(&other.patch));
        if c != std::cmp::Ordering::Equal { return c; }
        match (self.pre.is_empty(), other.pre.is_empty()) {
            (true, true) => std::cmp::Ordering::Equal,
            (true, false) => std::cmp::Ordering::Greater,
            (false, true) => std::cmp::Ordering::Less,
            (false, false) => {

struct Ver {

#[derive(Clone, Debug, Eq, PartialEq)]
enum PreId {
    Num(u64),
    Str(String),
}

impl Ord for PreId {
    fn cmp(&self, other: &Self) -> std::cmp::Ordering {
        use std::cmp::Ordering::*;
        match (self, other) {
            (PreId::Num(a), PreId::Num(b)) => a.cmp(b),
            (PreId::Str(a), PreId::Str(b)) => a.cmp(b),
            (PreId::Num(_), PreId::Str(_)) => Less,
            (PreId::Str(_), PreId::Num(_)) => Greater,
        }
    }
}
impl PartialOrd for PreId {
    fn partial_cmp(&self, other: &Self) -> Option<std::cmp::Ordering> {
        Some(self.cmp(other))
    }
}

impl Ord for Ver {
    fn cmp(&self, other: &Self) -> std::cmp::Ordering {
        use std::cmp::Ordering::*;
        let c = self.major.cmp(&other.major)
            .then(self.minor.cmp(&other.minor))
            .then(self.patch.cmp(&other.patch));
        if c != Equal { return c; }
        match (self.pre.is_empty(), other.pre.is_empty()) {
            (true, true) => Equal,
            (true, false) => Greater,
            (false, true) => Less,
            (false, false) => {
                for (a, b) in self.pre.iter().zip(other.pre.iter()) {
                    let c = a.cmp(b);
                    if c != Equal { return c; }
                }
                self.pre.len().cmp(&other.pre.len())
            }
        }
    }
}

    fn partial_cmp(&self, other: &Self) -> Option<std::cmp::Ordering> {
        Some(self.cmp(other))
    }

#[derive(Clone)]
enum Con {
    Eq(Ver), Gte(Ver), Gt(Ver), Lte(Ver), Lt(Ver), Caret(Ver), Tilde(Ver),
}

fn parse_cons(s: &str) -> Vec<Con> {
    s.split(',').map(|p| {
        let p = p.trim();
        if p.starts_with(">=") { Con::Gte(Ver::parse(&p[2..])) }
        else if p.starts_with('>') { Con::Gt(Ver::parse(&p[1..])) }
        else if p.starts_with("<=") { Con::Lte(Ver::parse(&p[2..])) }
        else if p.starts_with('<') { Con::Lt(Ver::parse(&p[1..])) }
        else if p.starts_with('=') { Con::Eq(Ver::parse(&p[1..])) }
        else if p.starts_with('^') { Con::Caret(Ver::parse(&p[1..])) }
        else if p.starts_with('~') { Con::Tilde(Ver::parse(&p[1..])) }
        else { Con::Eq(Ver::parse(p)) }
    }).collect()
}

fn sat(v: &Ver, c: &Con) -> bool {
    match c {
        Con::Eq(t) => v == t,
        Con::Gte(t) => v >= t,
        Con::Gt(t) => v > t,
        Con::Lte(t) => v <= t,
        Con::Lt(t) => v < t,
        Con::Caret(t) => {
            if v < t { return false; }
            let upper = if t.major != 0 {
                Ver { major: t.major+1, minor: 0, patch: 0, pre: vec![] }
            } else if t.minor != 0 {
                Ver { major: 0, minor: t.minor+1, patch: 0, pre: vec![] }
            } else {
                Ver { major: 0, minor: 0, patch: t.patch+1, pre: vec![] }
            };
            v < &upper
        }
        Con::Tilde(t) => {
            v >= t && v < &Ver { major: t.major, minor: t.minor+1, patch: 0, pre: vec![] }
        }
    }
}

fn sat_all(v: &Ver, cs: &[Con]) -> bool {
    cs.iter().all(|c| sat(v, c))
}

// ===== JSON writer with ordered keys =====

fn write_json(w: &mut dyn Write, val: &JsonVal, indent: usize, current: usize) {
    match val {
        JsonVal::Null => write!(w, "null").unwrap(),
        JsonVal::Bool(b) => write!(w, "{}", if *b {"true"} else {"false"}).unwrap(),
        JsonVal::Int(n) => write!(w, "{}", n).unwrap(),
        JsonVal::Float(f) => write!(w, "{}", f).unwrap(),
        JsonVal::Str(s) => {
            write!(w, "\"{}\"", s.replace('\\', "\\\\").replace('"', "\\\"")).unwrap();
        }
        JsonVal::Arr(items) => {
            if items.is_empty() {
                write!(w, "[]").unwrap();
            } else {
                write!(w, "[\n").unwrap();
                for (i, item) in items.iter().enumerate() {
                    for _ in 0..current+indent { write!(w, " ").unwrap(); }
                    write_json(w, item, indent, current+indent);
                    if i < items.len()-1 { write!(w, ",").unwrap(); }
                    write!(w, "\n").unwrap();
                }
                for _ in 0..current { write!(w, " ").unwrap(); }
                write!(w, "]").unwrap();
            }
        }
        JsonVal::Obj(pairs) => {
            if pairs.is_empty() {
                write!(w, "{{}}").unwrap();
            } else {
                write!(w, "{{\n").unwrap();
                for (i, (k, v)) in pairs.iter().enumerate() {
                    for _ in 0..current+indent { write!(w, " ").unwrap(); }
                    write!(w, "\"{}\": ", k).unwrap();
                    write_json(w, v, indent, current+indent);
                    if i < pairs.len()-1 { write!(w, ",").unwrap(); }
                    write!(w, "\n").unwrap();
                }
                for _ in 0..current { write!(w, " ").unwrap(); }
                write!(w, "}}").unwrap();
            }
        }
    }
}

enum JsonVal {

fn fmt_float(val: f64, prec: usize) -> String {
    // Round to prec decimal places
    let factor = 10f64.powi(prec as i32);
    let rounded = (val * factor).round() / factor;
    // Format with enough precision, then trim trailing zeros
    let s = format!("{:.prec$}", rounded, prec = prec);
    // Trim trailing zeros after decimal point, but keep at least one digit after '.'
    if s.contains('.') {
        let trimmed = s.trim_end_matches('0');
        if trimmed.ends_with('.') {
            format!("{}0", trimmed)
        } else {
            trimmed.to_string()
        }
    } else {
        s
    }
}

        dts.sort();
        JsonVal::Obj(vec![
            ("name".into(), JsonVal::Str(name.clone())),
            ("version".into(), JsonVal::Str(rv.version.clone())),
            ("license".into(), JsonVal::Str(rv.license.clone())),
            ("license_category".into(), JsonVal::Str(cat)),
            ("depth".into(), JsonVal::Int(*depth as i64)),
            ("direct_dependencies".into(), JsonVal::Arr(dd.into_iter().map(JsonVal::Str).collect())),
            ("dependents".into(), JsonVal::Arr(dts.into_iter().map(JsonVal::Str).collect())),
        ])
    }).collect();

    let bo_strs: Vec<String> = build_order.iter()
        .map(|n| format!("{}@{}", n, resolved[n].version))
        .collect();

    // ===== Vulnerability report =====
    let mut max_score: f64 = 0.0;
    let mut total_vuln = 0;
    let mut vuln_entries: Vec<(String, f64, f64)> = Vec::new();
    for n in &names {
        let e = round(eff[n], prec);
        if e > 0.0 {
            total_vuln += 1;
            let b = round(resolved[n].vulnerability_score, prec);
            vuln_entries.push((n.clone(), b, e));
            if e > max_score { max_score = e; }
        }
    }
    vuln_entries.sort_by(|a, b| b.2.partial_cmp(&a.2).unwrap().then(a.0.cmp(&b.0)));

    let crit_pkgs: Vec<JsonVal> = vuln_entries.iter().map(|(name, base, eff_s)| {
        JsonVal::Obj(vec![
            ("name".into(), JsonVal::Str(name.clone())),
            ("version".into(), JsonVal::Str(resolved[name].version.clone())),
            ("base_score".into(), JsonVal::Float(fmt_float(*base, prec))),
            ("effective_score".into(), JsonVal::Float(fmt_float(*eff_s, prec))),
        ])
